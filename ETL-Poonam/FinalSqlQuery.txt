#DROP TABLE data_analyst;
CREATE TABLE data_analyst (
  id SERIAL,
  job_title VARCHAR(1000),
  salary_estimate VARCHAR(100),
  industry varchar(200),
  location VARCHAR(100),
  state varchar(1000),
  city varchar(200),
  lower float,
  upper float,
  average_sal float,
  PRIMARY KEY (id)
)

#DROP TABLE data_analyst_final;

  id SERIAL,
  job_title VARCHAR(1000),
  state varchar(1000),
  city varchar(200),
  average_sal float,
  PRIMARY KEY (id)
